#!/usr/bin/env groovy

pipeline {

    agent {
        label 'java'
    }

    stages {
        stage('clone') {
            steps {
                echo 'cloning...'
                checkout scm
            }
        }
        stage('compile') {
            steps {
                echo "start compiling"
				sh script: 'javac HelloWorld.java'
				echo "stop compiling"
            }
        }
    }
}
